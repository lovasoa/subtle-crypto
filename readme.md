### Usage

```js
const SubtleCrypto = require('subtle-crypto')

// If running in the browser
SubtleCrypto === window.SubtleCrypto

// If running in node
SubtleCrypto = {
  encrypt,
  decrypt,
  sign,
  verify,
  digest,
  generateKey,
  deriveKey,
  deriveBits,
  importKey,
  exportKey,
  wrapKey,
  unwrapKey
}
```
